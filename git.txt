git init - создает локальный репозиторий
(указывает системе git, что нужно отслеживать изменения в текущей папке)

git add 'путь  до файла' 'или название файла'- добавляет файл к текущей версии
git commit - создвет версию на основе git add
git commit -m 'сообщение'
git log - список версий (коммитов)
git log --oneline - сокращенный список версий
git status - отображает изменения которые произошли после последней версии
git remote  - работа с удаленным репозиторием
git remote add <название> <url>- добавление привязка удаленного репозитория
git remote add origin git@github.com:Nurgazy1990/didactic-robot.git
git remote -v - список удаленных репозиториев
git remote rm <название> - удаление привязки
git remote rename <старое название> <новое название>
git remote rename origin main
git pushh название_репозитория название_ветки - отправка локальных изменений на удаленный
репозиторий
git pull - название_репозитория название_ветки - получение изменений из
удаленного репозитория
git reset - откат изменений или версий
git reset - откатывет все изменения
git reset <название файла>
git reset HEAD~кол-во - откатывает указанное количество последних коммитов (версий)
git diff - построчное отображение изменений
git diff <название файла> - просмотр изменений в конкретном файле
git branch - просмотр имеющихся веток
git branch <название ветки> - создание ветки
git branch --delete <название ветки> - удаление ветки
git checkout <название ветки> - переход на другую ветку
git checkout -b <название ветки> - создание ветки и переход на нее
